
import React, { useState, useRef } from 'react';
import { useNavigate } from 'react-router-dom';
import { useStore } from '../context/StoreContext';
import { User, Dog, Award, Heart, Camera, Check } from 'lucide-react';

const Onboarding: React.FC = () => {
  const { currentUser, completeOnboarding } = useStore();
  const navigate = useNavigate();
  const fileInputRef = useRef<HTMLInputElement>(null);

  const [step, setStep] = useState(1);
  const [role, setRole] = useState<'breeder' | 'owner' | 'enthusiast' | null>(null);
  
  const [formData, setFormData] = useState({
      bio: '',
      location: '',
      avatar: '',
      kennelName: '',
      kennelEstablished: '',
      experienceLevel: 'First Time'
  });

  if (!currentUser) {
      navigate('/login');
      return null;
  }

  const handleAvatarUpload = (e: React.ChangeEvent<HTMLInputElement>) => {
      const file = e.target.files?.[0];
      if (file) {
          const reader = new FileReader();
          reader.onloadend = () => {
              setFormData({ ...formData, avatar: reader.result as string });
          };
          reader.readAsDataURL(file);
      }
  };

  const handleFinish = () => {
      if (!role) return;
      
      const extraData: any = {
          bio: formData.bio,
          location: formData.location
      };

      if (role === 'breeder') {
          extraData.kennelName = formData.kennelName;
          extraData.breederInfo = {
              kennelEstablished: formData.kennelEstablished,
              clubs: [],
              website: ''
          };
      } else if (role === 'owner') {
          extraData.ownerInfo = {
              experienceLevel: formData.experienceLevel,
              activities: []
          };
      }

      completeOnboarding(currentUser.id, role, formData.avatar, extraData);
      navigate('/');
  };

  return (
    <div className="min-h-screen bg-gray-50 flex flex-col items-center pt-12 px-4 sm:px-6 lg:px-8">
        <div className="max-w-3xl w-full space-y-8">
            <div className="text-center">
                <h1 className="text-3xl font-extrabold text-gray-900">Welcome, {currentUser.name}!</h1>
                <p className="mt-2 text-lg text-gray-600">Let's set up your profile so you can connect with the community.</p>
            </div>
            
            {/* Progress Bar */}
            <div className="w-full bg-gray-200 rounded-full h-2.5">
                <div className="bg-brand-600 h-2.5 rounded-full transition-all duration-300" style={{ width: step === 1 ? '33%' : step === 2 ? '66%' : '100%' }}></div>
            </div>

            <div className="bg-white shadow rounded-lg p-8">
                
                {/* STEP 1: ROLE SELECTION */}
                {step === 1 && (
                    <div className="space-y-6 animate-fade-in">
                        <h2 className="text-xl font-bold text-gray-900">How do you identify?</h2>
                        <div className="grid grid-cols-1 md:grid-cols-3 gap-4">
                            <button 
                                onClick={() => setRole('breeder')}
                                className={`p-6 rounded-lg border-2 text-left transition hover:shadow-md flex flex-col items-center text-center ${role === 'breeder' ? 'border-brand-600 bg-brand-50 ring-1 ring-brand-600' : 'border-gray-200 hover:border-brand-300'}`}
                            >
                                <Award className="w-12 h-12 text-purple-600 mb-4" />
                                <h3 className="font-bold text-lg">Breeder</h3>
                                <p className="text-sm text-gray-500 mt-2">I actively breed Akitas and manage litters.</p>
                            </button>

                            <button 
                                onClick={() => setRole('owner')}
                                className={`p-6 rounded-lg border-2 text-left transition hover:shadow-md flex flex-col items-center text-center ${role === 'owner' ? 'border-brand-600 bg-brand-50 ring-1 ring-brand-600' : 'border-gray-200 hover:border-brand-300'}`}
                            >
                                <Dog className="w-12 h-12 text-brand-600 mb-4" />
                                <h3 className="font-bold text-lg">Owner</h3>
                                <p className="text-sm text-gray-500 mt-2">I own one or more Akitas as pets or show dogs.</p>
                            </button>

                             <button 
                                onClick={() => setRole('enthusiast')}
                                className={`p-6 rounded-lg border-2 text-left transition hover:shadow-md flex flex-col items-center text-center ${role === 'enthusiast' ? 'border-brand-600 bg-brand-50 ring-1 ring-brand-600' : 'border-gray-200 hover:border-brand-300'}`}
                            >
                                <Heart className="w-12 h-12 text-red-500 mb-4" />
                                <h3 className="font-bold text-lg">Enthusiast</h3>
                                <p className="text-sm text-gray-500 mt-2">I love the breed and want to learn more.</p>
                            </button>
                        </div>
                        <div className="flex justify-end pt-4">
                            <button 
                                disabled={!role}
                                onClick={() => setStep(2)}
                                className="bg-brand-600 text-white px-6 py-2 rounded-md font-medium hover:bg-brand-700 disabled:opacity-50 disabled:cursor-not-allowed"
                            >
                                Next
                            </button>
                        </div>
                    </div>
                )}

                {/* STEP 2: DETAILS */}
                {step === 2 && (
                     <div className="space-y-6 animate-fade-in">
                         <h2 className="text-xl font-bold text-gray-900">Tell us a bit more</h2>
                         
                         {role === 'breeder' && (
                             <div className="grid grid-cols-1 md:grid-cols-2 gap-4 bg-purple-50 p-4 rounded-lg border border-purple-100">
                                 <div>
                                     <label className="block text-sm font-medium text-gray-700">Kennel Name</label>
                                     <input type="text" value={formData.kennelName} onChange={e => setFormData({...formData, kennelName: e.target.value})} className="mt-1 block w-full border border-gray-300 rounded-md p-2" placeholder="e.g. Mountain Mist Akitas" />
                                 </div>
                                 <div>
                                     <label className="block text-sm font-medium text-gray-700">Year Established</label>
                                     <input type="text" value={formData.kennelEstablished} onChange={e => setFormData({...formData, kennelEstablished: e.target.value})} className="mt-1 block w-full border border-gray-300 rounded-md p-2" placeholder="e.g. 2010" />
                                 </div>
                             </div>
                         )}

                         {role === 'owner' && (
                              <div className="bg-green-50 p-4 rounded-lg border border-green-100">
                                  <label className="block text-sm font-medium text-gray-700">Experience Level</label>
                                  <select value={formData.experienceLevel} onChange={e => setFormData({...formData, experienceLevel: e.target.value})} className="mt-1 block w-full border border-gray-300 rounded-md p-2">
                                      <option>First Time</option>
                                      <option>Intermediate</option>
                                      <option>Expert</option>
                                  </select>
                              </div>
                         )}

                         <div>
                             <label className="block text-sm font-medium text-gray-700">Location</label>
                             <input type="text" value={formData.location} onChange={e => setFormData({...formData, location: e.target.value})} className="mt-1 block w-full border border-gray-300 rounded-md p-2" placeholder="City, State" />
                         </div>
                         <div>
                             <label className="block text-sm font-medium text-gray-700">Bio</label>
                             <textarea rows={3} value={formData.bio} onChange={e => setFormData({...formData, bio: e.target.value})} className="mt-1 block w-full border border-gray-300 rounded-md p-2" placeholder="Tell the community about yourself..." />
                         </div>

                         <div className="flex justify-between pt-4">
                            <button onClick={() => setStep(1)} className="text-gray-500 font-medium px-4 py-2">Back</button>
                            <button onClick={() => setStep(3)} className="bg-brand-600 text-white px-6 py-2 rounded-md font-medium hover:bg-brand-700">Next</button>
                        </div>
                     </div>
                )}

                {/* STEP 3: AVATAR */}
                {step === 3 && (
                     <div className="space-y-6 animate-fade-in text-center">
                        <h2 className="text-xl font-bold text-gray-900">Upload a Profile Photo</h2>
                        
                        <div className="flex flex-col items-center justify-center">
                            <div 
                                className="relative w-32 h-32 rounded-full bg-gray-100 overflow-hidden border-4 border-white shadow-lg cursor-pointer hover:opacity-90 transition"
                                onClick={() => fileInputRef.current?.click()}
                            >
                                {formData.avatar ? (
                                    <img src={formData.avatar} alt="Profile" className="w-full h-full object-cover" />
                                ) : (
                                    <div className="w-full h-full flex items-center justify-center text-gray-400">
                                        <Camera className="w-10 h-10" />
                                    </div>
                                )}
                            </div>
                            <button 
                                onClick={() => fileInputRef.current?.click()}
                                className="mt-4 text-sm text-brand-600 font-medium hover:text-brand-800"
                            >
                                {formData.avatar ? 'Change Photo' : 'Upload Photo'}
                            </button>
                            <input type="file" ref={fileInputRef} className="hidden" accept="image/*" onChange={handleAvatarUpload} />
                        </div>

                        <div className="flex justify-between pt-8 border-t mt-8">
                            <button onClick={() => setStep(2)} className="text-gray-500 font-medium px-4 py-2">Back</button>
                            <button 
                                onClick={handleFinish}
                                className="bg-green-600 text-white px-8 py-3 rounded-md font-bold hover:bg-green-700 shadow-lg flex items-center"
                            >
                                <Check className="w-5 h-5 mr-2" /> Complete Setup
                            </button>
                        </div>
                     </div>
                )}
            </div>
        </div>
    </div>
  );
};

export default Onboarding;
