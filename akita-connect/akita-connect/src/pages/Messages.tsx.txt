import React, { useState, useEffect } from 'react';
import { useStore } from '../context/StoreContext';
import { useSearchParams } from 'react-router-dom';

const Messages: React.FC = () => {
  const { messages, currentUser, users, sendMessage } = useStore();
  const [searchParams] = useSearchParams();
  const [selectedUserId, setSelectedUserId] = useState<string>('');
  const [inputText, setInputText] = useState('');

  // Initialize selection from URL or default to first available user
  useEffect(() => {
      const recipientParam = searchParams.get('recipient');
      if (recipientParam && users.find(u => u.id === recipientParam)) {
          setSelectedUserId(recipientParam);
      } else if (!selectedUserId && users.length > 1) {
          // Default to someone other than me
          const other = users.find(u => u.id !== currentUser.id);
          if(other) setSelectedUserId(other.id);
      }
  }, [searchParams, users, currentUser.id, selectedUserId]);

  // Filter messages between current user and selected user
  const conversation = messages.filter(
      m => (m.senderId === currentUser.id && m.receiverId === selectedUserId) ||
           (m.senderId === selectedUserId && m.receiverId === currentUser.id)
  ).sort((a,b) => a.timestamp - b.timestamp);

  const selectedUser = users.find(u => u.id === selectedUserId);

  const handleSend = (e: React.FormEvent) => {
      e.preventDefault();
      if(!inputText.trim() || !selectedUserId) return;
      sendMessage(selectedUserId, inputText);
      setInputText('');
  }

  return (
      <div className="bg-white shadow rounded-lg overflow-hidden h-[600px] flex">
          {/* Sidebar */}
          <div className="w-1/3 border-r border-gray-200 flex flex-col">
              <div className="p-4 border-b">
                  <h2 className="font-bold text-lg">Messages</h2>
              </div>
              <div className="flex-1 overflow-y-auto">
                  {users.filter(u => u.id !== currentUser.id).map(user => (
                      <div 
                        key={user.id} 
                        onClick={() => setSelectedUserId(user.id)}
                        className={`p-4 flex items-center cursor-pointer hover:bg-gray-50 ${selectedUserId === user.id ? 'bg-brand-50 border-l-4 border-brand-500' : ''}`}
                      >
                          <img src={user.avatar} className="w-10 h-10 rounded-full object-cover" alt={user.name} />
                          <div className="ml-3">
                              <p className="font-medium text-gray-900">{user.name}</p>
                              <p className="text-xs text-gray-500">{user.kennelName || user.role}</p>
                          </div>
                      </div>
                  ))}
              </div>
          </div>

          {/* Chat Area */}
          <div className="flex-1 flex flex-col">
              <div className="p-4 border-b bg-gray-50 flex items-center">
                   {selectedUser ? (
                       <>
                        <img src={selectedUser.avatar} className="w-8 h-8 rounded-full mr-2 object-cover" alt={selectedUser.name} />
                        <span className="font-bold">{selectedUser.name}</span>
                       </>
                   ) : <span className="text-gray-500">Select a conversation</span>}
              </div>
              <div className="flex-1 p-4 overflow-y-auto space-y-4 bg-gray-100">
                  {(!conversation.length && selectedUser) && <p className="text-center text-gray-400 text-sm">No messages yet. Say hello!</p>}
                  {conversation.map(msg => {
                      const isMe = msg.senderId === currentUser.id;
                      return (
                          <div key={msg.id} className={`flex ${isMe ? 'justify-end' : 'justify-start'}`}>
                              <div className={`max-w-xs px-4 py-2 rounded-lg shadow-sm ${isMe ? 'bg-brand-600 text-white rounded-br-none' : 'bg-white text-gray-800 rounded-bl-none'}`}>
                                  {msg.content}
                              </div>
                          </div>
                      )
                  })}
              </div>
              <div className="p-4 bg-white border-t">
                  <form onSubmit={handleSend} className="flex space-x-2">
                      <input 
                        type="text" 
                        className="flex-1 border border-gray-300 rounded-full px-4 py-2 focus:ring-brand-500 focus:outline-none disabled:bg-gray-100"
                        placeholder="Type a message..."
                        value={inputText}
                        onChange={e => setInputText(e.target.value)}
                        disabled={!selectedUser}
                      />
                      <button type="submit" disabled={!selectedUser} className="bg-brand-600 text-white px-4 py-2 rounded-full hover:bg-brand-700 disabled:opacity-50">Send</button>
                  </form>
              </div>
          </div>
      </div>
  );
};

export default Messages;
